def _objective(pack, n_items, master, cluster_weight=8.0):
  length_pen = pack.x_used  # 1순위 — 그대로
  unfit_pen = (n_items - pack.fitted_count) * 1e6
  cluster_pen = _count_cluster_breaks(pack.placements, master) * cluster_weight  # 신규 2순위
  return length_pen + unfit_pen + cluster_pen

# cluster_weight=8.0 의 의미: 카테고리 break 1번 = 트럭 8 in 늘어난 것과 동일 페널티
# SA가 자동 균형: 8 in 절약 위해 break 1개 줄임 / break 1개 늘려 8 in 절약
최종 효과 예측 (L001 26ft 케이스):
• 장입 길이 19.42 → 20.0 ft (-3% 효율, but 7.6 ft buffer 여전히 충분)
• 카테고리 break 15회 → 6회 (-60%)
• 작업자 적재 시간 약 30-40% 단축 (실 측정 필요)
